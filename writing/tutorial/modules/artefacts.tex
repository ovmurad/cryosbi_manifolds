\begin{frame}\allowdisplaybreaks\frametitle{Artefacts}

\begin{itemize}
\item \mydef{Artefacts}=features of the embedding that do not exist in the data (clusters, holes, ``arms'', ``horseshoes'')
  
%\pause
\item[]
\item What to beware of when you compute an embedding
  \benum
\item algorithms that \myemph{claim to} choose $\epsilon$ automatically
  \pause
\item confirming the embedding is ``correct'' by visualization
  \item[\mybullet] tends to over-smooth, i.e. $\epsilon$ over-estimated
  \pause
\item \myblue{K-nn} (default in {\tt sk-learn}!) instead of \myblue{radius-neighbors}: tends to create clusters
  \pause
\item large variations in density: stretching the low densities, contracting the high
\item[\mybullet] subsample data to make it more uniform
  \pause
  \item \myemph{``horseshoes''} ($\Rightarrow\phi$ almost singular):
  \item[\mybullet] select the e-vectors (coming next)
  \eenum
\item[]
\item Popular heuristics: LLE, t-SNE, UMAP, neural networks more prone to artefacts
  \end{itemize}
\end{frame}
