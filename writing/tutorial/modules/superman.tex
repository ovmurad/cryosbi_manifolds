\begin{frame}{Implementation: \superman vs. \sklearn}
	\begin{itemize}
		\item Performance:
		\begin{itemize}
			\item \superman: fast, real-time or semi-real-time
			\item \sklearn: slower; may not complete on large data sets
		\end{itemize}
		\item Statistical and geometric features in SuperMan:
		\begin{itemize}
			\item neighbor selection ($k$-NN, radius $\eps$)
			\item Riemannian metric estimation ($\rmetric*$)
			\item intrinsic dimensionality estimation
			\item other diagnostics and elementary statistics
		\end{itemize}
	\end{itemize}
\end{frame}
